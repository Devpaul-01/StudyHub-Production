/** * ============================================================================ * ENHANCED RESOURCE VIEWER V2 * Single resource display with horizontal scroll + auto-play * ============================================================================ */ class ResourceViewerV2 { constructor() { this.currentIndex = 0; this.resources = []; this.autoplayInterval = null; this.createModal(); } createModal() { const modal = document.createElement('div'); modal.id = 'resource-viewer-v2'; modal.className = 'resource-viewer-v2 hidden'; modal.innerHTML = `
1 / 1
💡 Swipe or use arrow keys to navigate
`; document.body.appendChild(modal); // Add CSS this.addStyles(); // Setup keyboard navigation document.addEventListener('keydown', (e) => { if (!modal.classList.contains('hidden')) { if (e.key === 'ArrowLeft') this.previous(); if (e.key === 'ArrowRight') this.next(); if (e.key === 'Escape') this.close(); } }); // Setup touch/swipe this.setupSwipe(); } addStyles() { const style = document.createElement('style'); style.textContent = ` .resource-viewer-v2 { position: fixed; inset: 0; z-index: 10000; display: flex; align-items: center; justify-content: center; } .resource-viewer-v2.hidden { display: none; } .viewer-overlay { position: absolute; inset: 0; background: rgba(0, 0, 0, 0.95); } .viewer-container { position: relative; width: 90vw; max-width: 1200px; height: 85vh; display: flex; flex-direction: column; gap: 1rem; } .viewer-close { position: absolute; top: -3rem; right: 0; background: rgba(255, 255, 255, 0.1); border: none; border-radius: 50%; width: 40px; height: 40px; cursor: pointer; color: white; display: flex; align-items: center; justify-content: center; transition: background 0.2s; } .viewer-close:hover { background: rgba(255, 255, 255, 0.2); } .viewer-content { flex: 1; display: flex; align-items: center; justify-content: center; overflow: hidden; border-radius: 12px; background: #000; } .viewer-content img, .viewer-content video { max-width: 100%; max-height: 100%; object-fit: contain; border-radius: 8px; } .viewer-controls { display: flex; align-items: center; justify-content: space-between; padding: 1rem; background: rgba(255, 255, 255, 0.1); backdrop-filter: blur(10px); border-radius: 12px; } .viewer-btn { background: rgba(255, 255, 255, 0.15); border: none; border-radius: 8px; width: 44px; height: 44px; cursor: pointer; color: white; display: flex; align-items: center; justify-content: center; transition: background 0.2s; } .viewer-btn:hover { background: rgba(255, 255, 255, 0.25); } .viewer-btn:disabled { opacity: 0.3; cursor: not-allowed; } .viewer-info { display: flex; align-items: center; gap: 1rem; color: white; font-size: 0.95rem; } .viewer-autoplay { background: rgba(255, 255, 255, 0.15); border: none; padding: 0.5rem 1rem; border-radius: 8px; color: white; cursor: pointer; transition: background 0.2s; } .viewer-autoplay:hover { background: rgba(255, 255, 255, 0.25); } .viewer-autoplay.active { background: var(--primary); } .viewer-hint { text-align: center; color: rgba(255, 255, 255, 0.6); font-size: 0.875rem; padding: 0.5rem; } @media (max-width: 768px) { .viewer-container { width: 95vw; height: 90vh; } } `; document.head.appendChild(style); } setupSwipe() { const content = document.getElementById('viewer-content'); let startX = 0; let startY = 0; content.addEventListener('touchstart', (e) => { startX = e.touches[0].clientX; startY = e.touches[0].clientY; }, { passive: true }); content.addEventListener('touchend', (e) => { const endX = e.changedTouches[0].clientX; const endY = e.changedTouches[0].clientY; const diffX = startX - endX; const diffY = startY - endY; // Horizontal swipe if (Math.abs(diffX) > Math.abs(diffY) && Math.abs(diffX) > 50) { if (diffX > 0) this.next(); else this.previous(); } }, { passive: true }); } open(resources, startIndex = 0) { if (!resources || resources.length === 0) { showToast('No resources available', 'info'); return; } // Filter to media only this.resources = resources.filter(r => r.type === 'image' || r.type === 'video'); if (this.resources.length === 0) { showToast('No media resources available', 'info'); return; } this.currentIndex = Math.min(startIndex, this.resources.length - 1); this.render(); const modal = document.getElementById('resource-viewer-v2'); modal.classList.remove('hidden'); document.body.style.overflow = 'hidden'; } render() { const resource = this.resources[this.currentIndex]; const content = document.getElementById('viewer-content'); const counter = document.getElementById('viewer-counter'); content.innerHTML = ''; if (resource.type === 'image') { const img = document.createElement('img'); img.src = resource.url; img.alt = resource.filename || 'Image'; content.appendChild(img); } else if (resource.type === 'video') { const video = document.createElement('video'); video.src = resource.url; video.controls = true; video.autoplay = false; content.appendChild(video); } counter.textContent = `${this.currentIndex + 1} / ${this.resources.length}`; } next() { if (this.currentIndex < this.resources.length - 1) { this.currentIndex++; this.render(); } } previous() { if (this.currentIndex > 0) { this.currentIndex--; this.render(); } } toggleAutoplay() { const btn = document.getElementById('autoplay-btn'); if (this.autoplayInterval) { clearInterval(this.autoplayInterval); this.autoplayInterval = null; btn.classList.remove('active'); btn.textContent = '▶️ Auto'; } else { this.autoplayInterval = setInterval(() => { if (this.currentIndex < this.resources.length - 1) { this.next(); } else { this.currentIndex = 0; this.render(); } }, 3000); btn.classList.add('active'); btn.textContent = '⏸️ Auto'; } } close() { const modal = document.getElementById('resource-viewer-v2'); modal.classList.add('hidden'); document.body.style.overflow = ''; if (this.autoplayInterval) { clearInterval(this.autoplayInterval); this.autoplayInterval = null; } } } // Initialize globally window.resourceViewerV2 = new ResourceViewerV2(); // Compatibility functions window.openPostResourceViewer = function(postId, startIndex, event) { if (event) event.preventDefault(); const post = document.getElementById(`post-${postId}`); if (!post) return; const resources = JSON.parse(post.dataset.resources || '[]'); resourceViewerV2.open(resources, startIndex); }; window.openCommentResourceViewer = function(commentId, startIndex, event) { if (event) event.preventDefault(); const comment = document.getElementById(`comment-card-${commentId}`); if (!comment) return; const resources = JSON.parse(comment.dataset.resources || '[]'); resourceViewerV2.open(resources, startIndex); };
